public class PrintFile {
    private String printFileURL;

    public PrintFile() {
    }

    public PrintFile(String printFileURL) {
        this.printFileURL = printFileURL;
    }

    public String getPrintFileURL() {
        return printFileURL;
    }

    public void setPrintFileURL(String printFileURL) {
        this.printFileURL = printFileURL;
    }
}
